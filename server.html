// server.js require('dotenv').config(); const express = require('express'); const bodyParser = require('body-parser'); const cors = require('cors'); const app = express(); // Middleware setup app.use(cors()); // Enables CORS app.use(bodyParser.json()); // Parses JSON data in requests // Basic Route to Test the Server app.get('/', (req, res) => { res.send('Booking server is running!'); }); // Start the server const PORT = process.env.PORT || 5000; // Port from environment variables or default app.listen(PORT, () => { console.log(`Server is running on port ${PORT}`); });
